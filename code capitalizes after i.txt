#include<iostream>
#include<cstring>
using namespace std;
int main()
{
	char str[10000];
	cin.getline(str, 10000);
	int i = 0;
	for (;i < strlen(str);i++)
	{
		if (str[i] == 'i')
		{
			if (str[i + 1] >= 'a'&&str[i + 1] <= 'z')
			{
				str[i + 1] -= 'a' - 'A';
			}
		}
	}
		cout << str << endl;
	

}
